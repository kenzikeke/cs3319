--build tables
SHOW TABLES;
CREATE TABLE product( productID CHAR(2) PRIMARY KEY NOT NULL, description VARCHAR(20) NOT NULL, cost FLOAT NOT NULL, instock INT NOT NULL);
CREATE TABLE agent( agentID CHAR(2) PRIMARY KEY NOT NULL, firstName CHAR(30) NOT NULL, lastName CHAR(30) NOT NULL, city CHAR(15) NOT NULL, commissionPercent CHAR NOT NULL);
CREATE TABLE customer(customerID CHAR(2) PRIMARY KEY NOT NULL, firstName VARCHAR(30) NOT NULL, lastName VARCHAR(30) NOT NULL, city VARCHAR(15) NOT NULL, phone CHAR NOT NULL, servedBy CHAR(2) NOT NULL, FOREIGN KEY(servedBy) REFERENCES agent(agentID));
CREATE TABLE purchased(customerID CHAR(2) NOT NULL, productID CHAR(2) NOT NULL, number INT NOT NULL, FOREIGN KEY(customerID) REFERENCES customer(customerID),FOREIGN KEY(productID) REFERENCES product(productID));
